    # VueGridLayout Component

    # Example usage

    <template>
    <WidthProvider>
        <template v-slot:default="{width}">
            <div class="content" ref="offsetParent">
                <VueGridLayout
                    :layout="layout"
                    :rowHeight="20"
                    :cols="12"
                    :width="width"
                >
                    <template v-slot:item="{i}">
                        <span class="text">{{i}}</span>
                    </template>
                </VueGridLayout>
            </div>
        </template>
    </WidthProvider>
</template>

<script>
    export default {
        name: 'GridLayout',
        data() {
            return {
                layout: [
                    {
                        i: "0",
                        x: 0,
                        y: 0,
                        w: 2,
                        h: 5,
                        draggable: true,
                        resizable: true
                    },
                    {
                        i: "1",
                        x: 2,
                        y: 0,
                        w: 2,
                        h: 2,
                        draggable: true,
                        resizable: true
                    },
                    {
                        i: "static",
                        x: 4,
                        y: 0,
                        w: 2,
                        h: 4,
                        draggable: true,
                        resizable: true,
                        static: true
                    },
                    {
                        i: "3",
                        x: 6,
                        y: 0,
                        w: 2,
                        h: 2,
                        draggable: true,
                        resizable: true
                    },
                ],
            };
        },
    }

</script>
